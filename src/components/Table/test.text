        {col.sorter && (
                <div>
                  <button
                    onClick={() => {
                      const sortedData = [...data].sort(col.sorter);
                      setVisibleData(sortedData);
                    }}
                  >
                    ↑
                  </button>
                  <button 
                    onClick={() => {
                        const sortedData = [...data].sort(col.sorter).reverse();
                        setVisibleData(sortedData);
                    }}
                  >↓</button>
                </div>
              )}
